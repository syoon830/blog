var 키워드의 문제점

1. 변수 중복 선언 허용

이미 선언된 변수의 값이 변경되는 부작용 발생
var x = 1;
var y = 1;

// var 키워드로 선언된 변수는 같은 스코프 내에서 중복 선언을 허용한다.
// 초기화문이 있는 변수 선언문은 자바스크립트 엔진에 의해 var 키워드가 없는 것처럼 동작한다.
var x = 100;
// 초기화가 없는 변수 선언문은 무시된다.
var y;

console.log(x); // 100
console.log(y); // 1

2. 함수 레벨 스코프

오로지 함수의 코드 블록만을 지역 스코프로 인정한다.
var x = 1;

if (true) {
    var x = 10;
}
console.log(x); // 10

var i = 10;

for (var i = 0; i < 5; i++) {
    console.log(i); // 0 1 2 3 4 5
}
console.log(i); // 5

3. 변수 호이스팅

console.log(foo);

foo = 123;

console.log(foo);

var foo;

let 키워드

1. 변수 중복 선언 금지

var foo = 123;
var foo = 456;

let bar = 123;
let bar = 123; // SyntaxError

2. 블록 레벨 스코프

let foo = 1;

{
    let foo = 2;
    let bar = 3;
}
console.log(foo); // 1
console.log(bar); // ReferenceError

3. 변수 호이스팅

var 키워드는 런타임 이전에 선언 단계와 초기화 단계가 한번에 진행됨
let 키워드는 런타임 이전에 선언 단계와 초기화 단계가 분리되어 진행
let 키워드로 선언한 변수는 스코프의 시작 지점부터 초기화 단계 시작 지점(변수 선언문)까지 변수를 참조할 수 없다. 스코프의 시작 지점부터 초기화 시작 시작 지점까지 변수를 참조할 수 없는 구간을 일시적 사각지대(TDZ)라고 부른다.
console.log(foo); // ReferenceError
let foo;
let 키워드는 호이스팅이 발생하지 않는 것처럼 보인다. 하지만 그렇지 않다. 아래 코드를 보면 호이스팅이 발생하지 않는다면 1이 출력 되어야 하는데 참조 에러가 발생한다.
자바스크립트는 모든 선언을 호이스팅한다. (단, let, const class를 사용한 선언문은 호이스팅이 발생하지 않는 것처럼 동작한다.)
간단히 말해서, let과 const는 선언된 블록 스코프 내에서 선언 지점 이전에는 "존재하지 않는 것"처럼 취급된다. 이 구간을 TDZ라고 한다.
let foo = 1;
{
    console.log(foo); // TDZ
    let foo = 2; // 초기화 시작 지점
}

4. 전역 객체와 let

전역으로 선언한 var는 window 객체에 프로퍼티가 된다.
let 키워드로 선언한 전연 변수는 window 객체에 프로퍼티가 아니다.
let 전역 변수는 보이지 않는 개념적인 블록내에 존재한다.
var x = 1;
y = 2;

function foo(){}

console.log(window.x);
console.log(window.y);
console.log(window.foo);

-------

let x = 1;
console.log(window.x); // undefined

const 키워드

1. 선언과 초기화

const foo = 1;

const foo; // SyntaxError

2. 재할당 금지

const foo = 1;
foo = 2; // TypeError

3. 상수

const TAX_RATE = 0.1;

let preTaxPrice = 100;

let afterTaxPrice = preTaxPrice + (preTaxPrice * TAX_RATE);

4. const 키워드와 객체

const 키워드로 선언된 변수에 원시 값을 할당한 경우 값을 변경할 수 없다. 하지만 객체를 할당한 경우 값을 변경할 수 있다.
그 이유는 원시 값은 변경 불가능한 값이지만
객체는 변경 가능한 값이기 때문에 재할당 없이 직접 변경이 가능하다.
const person = {
    name: 'Lee'
};

person.name = 'kim';

console.log(person.name); // kim
const는 재할당을 금지할 뿐 “불변”을 의미하지는 않는다.
결론: ES6를 사용한다면 var를 쓸 이유는 없다.
